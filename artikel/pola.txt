
        (function setupShareLinks() {
            const text = encodeURIComponent(ARTICLE_TITLE);
            const url = encodeURIComponent(CANON_URL);
            const via = encodeURIComponent(VIA_TWITTER);
            document.getElementById('share-wa').href = `https://wa.me/?text=${text}%20${url}`;
            document.getElementById('share-tw').href = `https://twitter.com/intent/tweet?text=${text}&url=${url}&via=${via}`;
            document.getElementById('share-fb').href = `https://www.facebook.com/sharer/sharer.php?u=${url}`;
            document.getElementById('share-li').href = `https://www.linkedin.com/sharing/share-offsite/?url=${url}`;
        })();
